project:
  name: "Cosmos 2.0 – Plataforma Espiritual com Astrologia e IA"
  type: "SaaS Web Application"
  platform: "Web (Mobile-first e PWA)"
  description: |
    Cosmos 2.0 é uma plataforma SaaS interativa que une astrologia, psicologia simbólica e IA
    para ajudar o usuário a se conhecer melhor. O sistema gera um mapa astral personalizado,
    interpreta sonhos com IA conversacional e guia o explorador espiritual por meio de conselhos diários,
    fases lunares e um diário simbólico. A experiência é gamificada, emocional e minimalista.

architecture:
  backend:
    framework: "Flask (Python)"
    services:
      - Authentication: "Replit Auth (Google, GitHub, Email/Senha) + perfis no PostgreSQL"
      - Database: "PostgreSQL (usuários, diário, gamificação)"
      - Cache: "Firestore ou Redis para trânsitos e consultas astrológicas"
      - AI: "OpenAI GPT (interpretação simbólica e persona onírica)"
      - Astrology API: "AstrologyAPI ou Swiss Ephemeris"
  frontend:
    framework: "React + TailwindCSS + Vite"
    libraries:
      - ui: "shadcn/ui, lucide-react"
      - animation: "framer-motion"
      - charts: "recharts (wheel chart)"
      - audio: "Web Audio API para feedback sonoro cósmico"
  hosting:
    environment: "Replit Deployments (PaaS) ou Google Cloud Run"
  secrets:
    - OPENAI_API_KEY
    - ASTRO_API_KEY
    - DATABASE_URL
    - GOOGLE_APPLICATION_CREDENTIALS_JSON

structure:
  folders:
    - /app.py
    - /api
    - /static/css
    - /static/js
    - /templates
    - /utils
    - /components
  key_files:
    - .replit: |
        run = "python app.py"
    - replit.nix: |
        { pkgs }: {
          deps = [ pkgs.python311 pkgs.nodejs pkgs.redis pkgs.ffmpeg ];
        }

modules:
  - name: "Cadastro e Autenticação"
    features:
      - Login com Replit Auth (Google, GitHub, Email/Senha)
      - Perfis salvos no PostgreSQL
      - Sessões persistentes com tokens seguros
      - Conta anônima temporária para onboarding rápido
  - name: "Onboarding Gamificado"
    features:
      - Fluxo interativo com storytelling cósmico
      - Campos: data, hora, local de nascimento (autocompletar via Google Maps API)
      - Feedback visual e sonoro conforme o signo solar
      - Preview parcial do mapa astral ao final do onboarding
  - name: "Dashboard – Meu Dia"
    features:
      - Exibe energia principal do dia com cards animados
      - Conselhos práticos baseados nos trânsitos
      - Sol e Lua atuais + “Humor cósmico coletivo”
      - Botão “Salvar” e “Compartilhar nas redes”
      - Cache offline PWA (últimos conselhos)
  - name: "Aba Sol & Lua"
    features:
      - Signos, graus e significados diários
      - Fases lunares com animação SVG
      - Próxima fase e recomendações energéticas
      - Ícones planetários interativos (Sol, Lua, Mercúrio, Vênus, Marte)
  - name: "Aba Meu Mapa Astral"
    features:
      - Wheel chart interativo com hover e zoom
      - Descrições de Sol, Lua, Ascendente e casas
      - Modo “Resumo Rápido” com 3 insights principais
      - Comparação “Hoje vs Mapa Natal”
  - name: "Diário de Sonhos (IA Conversacional)"
    features:
      - Chat com persona “IA Onírica” (voz e texto)
      - Entrada de voz via Whisper API
      - Interpretação com:
          - Tema central
          - Símbolos-chave
          - Mensagem do inconsciente
          - Ritual sugerido
      - Histórico de sonhos favoritos no PostgreSQL
  - name: "Gamificação e Progressão"
    features:
      - Selos e conquistas (“Semana da Clareza”, “Ciclo Lunar Completo”)
      - Constelação interativa (cada ação acende uma estrela)
      - Níveis simbólicos (“Mago Lunar”, “Alquimista Solar”)
      - Estatísticas cósmicas pessoais e lembretes
      - Notificações motivacionais (“O Sol sorri para o seu signo hoje 🌞”)
