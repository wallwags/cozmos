project:
  name: "Cosmos 2.0 – MVP Expandido (Híbrido)"
  type: "SaaS Web Application"
  platform: "Web (Mobile-first, PWA-ready)"
  short_description: >
    Plataforma espiritual que combina astrologia e IA para conselhos diários, mapa natal
    e interpretação de sonhos. Lançamento rápido com base robusta para ativar recursos de
    voz, cache e animações em iterações seguintes.

architecture:
  backend:
    framework: "Flask (Python)"
    primary_db: "PostgreSQL"      # perfis, diário, gamificação (core)
    cache: "Redis (ou Firestore TTL)" # trânsitos e dados voláteis (iteration-1)
    ai_service: "OpenAI GPT (texto) - core; Whisper integrado em iteration-1"
    astro_service: "AstrologyAPI (ou Swiss Ephemeris)"
    auth: "Replit Auth (Google/GitHub/Email) + sessions em DB"
  frontend:
    framework: "React + Vite + TailwindCSS"
    ui_libraries: ["shadcn/ui","lucide-react","recharts"]
    pwa_ready: true   # service worker scaffold (core) + full offline in iteration-1
  hosting:
    primary: "Replit Deployments"
    optional_scaling: "Google Cloud Run (for heavy tasks like batch ephemeris)"
  secrets:
    - OPENAI_API_KEY
    - ASTRO_API_KEY
    - DATABASE_URL
    - REDIS_URL
    - REPLIT_AUTH_KEYS
    - GOOGLE_MAPS_API_KEY

project_scope:
  goal: "Validar product-market fit com Explorador Espiritual mantendo caminho claro para features ricas."
  success_criteria:
    - 1000 MAU dentro de 3 meses (meta inicial)
    - Taxa de ativação (signup -> completar onboarding) >= 40%
    - Retenção 7 dias >= 20%

    name: "Dashboard – Meu Dia"
    priority: "core"
    deliverables:

  - id: pwa_serviceworker

dev_process:

  branching: "feature/* -> PR -> CI -> main"
  ci_cd:
    - run_tests: "unit tests for backend + frontend"
    - linting: "prettier + eslint + flake8"
    - auto_deploy: "main -> Replit Deployments after approval"
  checkpoints:
    - checkpoint_1: "Core features working end-to-end (auth, onboarding, map, dashboard, dreams text) - deployable"
    - checkpoint_2: "iteration-1 features (PWA, Redis, Whisper) integrated in staging"
    - checkpoint_3: "iteration-2 polish + analytics + performance tuning"

  - e2e_tests: signup -> onboarding -> dashboard -> dream analyze flow

  - load_test: simulate trânsitos API traffic

  - plan Redis/Cloud Run scaling thresholds
